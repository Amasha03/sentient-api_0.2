    // --- CONFIGURATION ---
    const API = ''; // Make sure this matches your Flask port
    let convId = genId(), convs = JSON.parse(localStorage.getItem('e3_convs') || '{}');
    let isRec = false, mediaRec = null, chunks = [], speechRec = null, curTranscript = '';
    let currentLanguage = 'english'; // Default language
    // ── INIT ──
    (function(){
    const user = JSON.parse(localStorage.getItem('emotion3_user') || '{"name":"User"}');
    document.getElementById('uName').textContent = user.name;
    document.getElementById('uAvatar').textContent = user.name.charAt(0).toUpperCase();
    renderList();
    greeting();
    })();
    function genId(){return 'c_'+Date.now()+'_'+Math.random().toString(36).slice(2,6)}
    // ── CONVERSATIONS ──
    function saveConvs(){localStorage.setItem('e3_convs',JSON.stringify(convs))}
    function renderList(){
    const el=document.getElementById('convList');el.innerHTML='';
    const ids=Object.keys(convs).reverse();
    if(!ids.length){el.innerHTML='<div style="font-size:.8rem;color:var(--muted);padding:8px 12px">No conversations yet</div>';return}
    ids.forEach(id=>{
    const d=document.createElement('div');
    d.className='conv-item'+(id===convId?' active':'');
    d.textContent=convs[id].title||'Untitled';
    d.onclick=()=>loadConv(id);
    el.appendChild(d);
    });
    }
    function loadConv(id){
    convId=id;document.getElementById('messages').innerHTML='';
    const c=convs[id];
    if(c&&c.msgs)c.msgs.forEach(m=>addBubble(m.role,m.content,m.itype,m.emotion,false));
    renderList();
    }
    function newConv(){
    convId=genId();document.getElementById('messages').innerHTML='';greeting();renderList();
    }
    function greeting(){
    addBubble('bot',"Hello! I'm <strong>E.motion³</strong>. 🌱<br>How are you feeling? Type or speak — I support English, Sinhala, and Tamil.",null,null,false);
    }
    // ── PIPELINE ANIMATION ──
    const PS=['ps-in','ps-mod','ps-llm','ps-tts'];
    function pReset(){PS.forEach(s=>document.getElementById(s).classList.remove('on'))}
    async function pAnimate(steps){
    pReset();
    for(const s of steps){
    const el = document.getElementById(s);
    if(el) el.classList.add('on');
    await new Promise(r=>setTimeout(r, 600));
    }
    }
    // ── UI BUBBLES ──
    function esc(s){return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')}
    function fmt(s){return s.replace(/\*\*([^*]+)\*\*/g,'<strong>$1</strong>')}
    function eEmoji(e){return{happy:'😊',sad:'😢',anxious:'😟',angry:'😠',disgust:'🤢',neutral:'😌'}[e.toLowerCase()]||'💭'}
    function addBubble(role,text,itype,emotion,anim=true){
    const wrap=document.getElementById('messages');
    const row=document.createElement('div');row.className='msg-row '+(role==='user'?'user':'bot');
    if(role==='user'&&itype){const t=document.createElement('div');t.className='itag';t.innerHTML=itype==='voice'?'🎙️ Voice':'💬 Text';row.appendChild(t)}
    const b=document.createElement('div');b.className='bubble '+(role==='user'?'user':'bot');
    b.innerHTML=role==='user'?esc(text):fmt(text);row.appendChild(b);
    if(role==='bot'&&emotion){const c=document.createElement('div');c.className='emotion-chip';c.innerHTML=eEmoji(emotion)+' Detected: <strong>'+emotion+'</strong>';row.appendChild(c)}
    if(!anim)b.style.animation='none';
    wrap.appendChild(row);wrap.scrollTop=wrap.scrollHeight;return row;
    }
    function showTyping(){
    const w=document.getElementById('messages');
    const r=document.createElement('div');r.className='typing-row';r.id='typing';
    r.innerHTML='<div class="typing-bub"><div class="dot"></div><div class="dot"></div><div class="dot"></div></div>';
    w.appendChild(r);w.scrollTop=w.scrollHeight;
    }
    function hideTyping(){const t=document.getElementById('typing');if(t)t.remove()}
    // ── CORE LOGIC: SEND TO PYTHON BACKEND ──
    async function sendText(){
    const inp=document.getElementById('userInput');
    const txt=inp.value.trim();if(!txt)return;
    inp.value='';document.getElementById('sendBtn').disabled=true;
    addBubble('user',txt,'text',null);
    pAnimate(['ps-in','ps-mod','ps-llm','ps-tts']);
    showTyping();
    try {
    const response = await fetch(`${API}/api/python/predict`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
    message: txt,
    language: currentLanguage,
    type: 'text'
    })
    });
    const data = await response.json();
    hideTyping();
    if (data.status === 'success') {
    addBubble('bot', data.reply_text, null, data.emotion);
    save('user', txt, 'text', null);
    save('bot', data.reply_text, 'text', data.emotion);
    // Play Audio from Base64
    if (data.reply_audio_base64) {
    const audio = new Audio(data.reply_audio_base64);
    audio.play();
    }
    } else {
    addBubble('bot', 'Error: ' + data.message);
    }
    } catch (e) {
    hideTyping();
    addBubble('bot', '⚠️ Connection failed. Is the Flask server running?');
    }
    document.getElementById('sendBtn').disabled=false;
    }
    // ── VOICE LOGIC ──
    async function toggleRec(){if(!isRec)await startRec();else stopRec()}
    async function startRec(){
    try{
    const stream=await navigator.mediaDevices.getUserMedia({audio:true});
    mediaRec=new MediaRecorder(stream);chunks=[];curTranscript='';
    mediaRec.ondataavailable=e=>{if(e.data.size>0)chunks.push(e.data)};
    mediaRec.onstop=handleStop;mediaRec.start();
    isRec=true;
    document.getElementById('micBtn').classList.add('rec');
    const SR=window.SpeechRecognition||window.webkitSpeechRecognition;
    if(SR){
    speechRec=new SR();speechRec.continuous=true;speechRec.interimResults=true;
    speechRec.onresult=e=>{
    let int='';
    for(let i=e.resultIndex;i<e.results.length;i++){
    if(e.results[i].isFinal) curTranscript+=e.results[i][0].transcript;
    else int+=e.results[i][0].transcript;
    }
    document.getElementById('userInput').placeholder=(curTranscript+int).trim() || 'Listening...';
    };
    speechRec.start();
    }
    }catch(e){alert('Mic access denied')}
    }
    function stopRec(){
    isRec=false;
    document.getElementById('micBtn').classList.remove('rec');
    if(speechRec){speechRec.stop()}
    if(mediaRec)mediaRec.stop();
    }
    async function handleStop(){
    const transcript = curTranscript.trim() || "Voice Input";
    addBubble('user', transcript, 'voice', null);
    showTyping();
    pAnimate(['ps-in','ps-mod','ps-llm','ps-tts']);
    // For simplicity, we send the transcript.
    // If you want to send actual audio file, you'd use FormData and 'type': 'voice'
    try {
    const response = await fetch(`${API}/api/python/predict`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
    message: transcript,
    language: currentLanguage,
    type: 'text' // Change to 'voice' if backend is ready to handle binary audio
    })
    });
    const data = await response.json();
    hideTyping();
    if(data.status === 'success'){
    addBubble('bot', data.reply_text, null, data.emotion);
    if(data.reply_audio_base64) new Audio(data.reply_audio_base64).play();
    }
    } catch(e) {
    hideTyping();
    }
    }
    // ── HELPERS ──
    function save(role,content,itype,emotion){
    if(!convs[convId])convs[convId]={title:content.slice(0,20),msgs:[]};
    convs[convId].msgs.push({role,content,itype,emotion});
    saveConvs();
